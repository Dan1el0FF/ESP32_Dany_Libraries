
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "Kinematics.h"



extern "C" void app_main() {

    float dh_parameters[3][4] = { 
        {90,10,90,0},
        {30,0,0,15},
        {45,0,0,20}};

    int joints=3;
    Kinematics scara;
    float pose[6];
    float coords[4]={0,18.16,36,0};
    float solutions[2][4];
    float best_solution[4];
    scara.setup(dh_parameters,joints);
    scara.calculate_FK(pose);
    scara.calculate_IK(coords,solutions);
    for(int i = 0;i<4;i++){
        best_solution[i]=solutions[scara.best_move(solutions)][i];
    }
    
    printf("theta_1: %f theta_2: %f Phi: %f Z: %f \n",best_solution[0],best_solution[1],best_solution[2],best_solution[3]);
    printf("X: %f Y: %f Z: %f roll: %f pitch: %f yaw: %f \n",pose[0],pose[1],pose[2],pose[3],pose[4],pose[5]);


}
