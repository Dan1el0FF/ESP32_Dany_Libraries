#include "GhostEncoder.h"
#include "stdio.h"

#define AVG_FRQZ 300
#define limit_switch_Z GPIO_NUM_16
GhostEncoder Z_step;

extern "C" void app_main(void){
    gpio_set_direction(limit_switch_Z,GPIO_MODE_INPUT);
    Z_step.setup(GPIO_NUM_19,GPIO_NUM_23);
    Z_step.setwaypoint(-1e6,AVG_FRQZ);
    bool LMS_Z=true;
    bool flag=true;
    while(flag){   
        LMS_Z = gpio_get_level(limit_switch_Z);
        if(LMS_Z!=0){
            Z_step.reset(); 
            flag=false;
        }
    }
}
