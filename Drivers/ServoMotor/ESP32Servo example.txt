//EXAMPLE ESP32SERVO 
#include <ESP32Servo.h>
#include <driver/gpio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#define up GPIO_NUM_17
#define down GPIO_NUM_16
#define escpin GPIO_NUM_18
extern "C" void app_main(void){

int vel=0;

gpio_set_direction(up,GPIO_MODE_INPUT);
gpio_set_direction(down,GPIO_MODE_INPUT);
ESP32Servo motor(0,escpin);
motor.setAngle(vel);
vTaskDelay(5000/portTICK_PERIOD_MS);

    while (true)
    {
        bool upstate,downstate;
        upstate=gpio_get_level(up);
        downstate=gpio_get_level(down);
        if(upstate) vel+=10;
        if(downstate) vel-=10;
        if(vel>180) vel=180;
        if(vel<0) vel=0;
        motor.setAngle(vel);
        vTaskDelay(100/portTICK_PERIOD_MS);
    }
}
