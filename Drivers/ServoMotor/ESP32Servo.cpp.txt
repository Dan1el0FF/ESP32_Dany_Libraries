//PWM_PIN_ACTUALIZADA_VERSION_SERVOZ_CONTROLLER
#include "ESP32Servo.h"

ESP32Servo::ESP32Servo(int channel_, int pin_){ //constructor inicializa parametros.
    channel=channel_;
    pin=pin_;
    configurePwm();
}
void ESP32Servo::configurePwm(){
    
    ledc_timer_config_t ledc_timer = {
        .speed_mode = LEDC_HIGH_SPEED_MODE,
        .duty_resolution = LEDC_TIMER_10_BIT, //resolucion de 0-1023
        .timer_num = LEDC_TIMER_0,
        .freq_hz = 50,
        .clk_cfg = LEDC_AUTO_CLK,
        .deconfigure = false,
    };
    ledc_timer_config(&ledc_timer); 

    ledc_channel = {
        .gpio_num = pin,
        .speed_mode = LEDC_HIGH_SPEED_MODE,
        .channel = (ledc_channel_t)channel,
        .intr_type = LEDC_INTR_DISABLE,
        .timer_sel = LEDC_TIMER_0,
        .duty = 0,
        .hpoint = 0,
        .flags = 0,
    };
    ledc_channel_config(&ledc_channel);

}

void ESP32Servo::setAngle(int angle_){  //setea el angulo entre 0-180 grados con una resolucion de 10 bits.

    if(angle_>=180)angle_ = 180;
    else if (angle_ <=0)angle_ = 0;
    
    int output = ((angle_ * 51) / 180) + 51;
    
    angle=output;
    
    ledc_set_duty(ledc_channel.speed_mode, ledc_channel.channel, angle);
    ledc_update_duty(ledc_channel.speed_mode, ledc_channel.channel);
}

int ESP32Servo::getAngle(){
    return angle;
}
